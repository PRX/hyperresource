def initialize(message, attrs={}) # @private
  self.response = attrs[:response]
  self.body = attrs[:body]

  ## Try to help out with the message
  if self.body
    if error = self.body['error']
      message = "#{message} (#{error})"
    end
  elsif self.response
    message = "#{message} (\"#{self.response.inspect}\")"
  end

  super(message, attrs)
end
